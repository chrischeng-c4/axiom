use crate::models::ArchiveReviewVerdict;
use crate::Result;
use std::path::Path;

/// Parse merge/archive review verdict from MERGE_REVIEW.md or ARCHIVE_REVIEW.md
///
/// Looks for checked boxes in the status section:
/// - [x] APPROVED → Approved
/// - [x] REVIEWED → Reviewed
/// - [x] REJECTED → Rejected
///
/// Also supports fallback formats:
/// - "Verdict: APPROVED" (without checkbox)
/// - "**Verdict**: APPROVED"
/// @spec projects/agentic-workflow/tech-design/core/interfaces/parser/archive_review.md#source
pub fn parse_archive_review_verdict(review_path: &Path) -> Result<ArchiveReviewVerdict> {
    if !review_path.exists() {
        return Ok(ArchiveReviewVerdict::Unknown);
    }

    let content = std::fs::read_to_string(review_path)?;
    let content_lower = content.to_lowercase();

    // Primary: Look for checked boxes (generated by MCP tool)
    if content_lower.contains("[x] approved") {
        return Ok(ArchiveReviewVerdict::Approved);
    }
    if content_lower.contains("[x] reviewed") || content_lower.contains("[x] needs fix") {
        return Ok(ArchiveReviewVerdict::Reviewed);
    }
    if content_lower.contains("[x] rejected") {
        return Ok(ArchiveReviewVerdict::Rejected);
    }

    // Fallback: Look for "verdict: X" or "**verdict**: X" patterns (LLM direct output)
    for line in content_lower.lines() {
        let line = line.trim();
        // Match patterns like "verdict: approved", "**verdict**: approved", "verdict:** approved"
        if line.contains("verdict") {
            if line.contains("approved") && !line.contains("not approved") {
                return Ok(ArchiveReviewVerdict::Approved);
            }
            if line.contains("reviewed") || line.contains("needs fix") || line.contains("needsfix")
            {
                return Ok(ArchiveReviewVerdict::Reviewed);
            }
            if line.contains("rejected") {
                return Ok(ArchiveReviewVerdict::Rejected);
            }
        }
    }

    // If no verdict found, it's unknown
    Ok(ArchiveReviewVerdict::Unknown)
}

/// Get the review file path, preferring review_merge.md over legacy names
/// @spec projects/agentic-workflow/tech-design/core/interfaces/parser/archive_review.md#source
pub fn get_review_path(change_dir: &Path) -> std::path::PathBuf {
    let review_merge = change_dir.join("review_merge.md");
    if review_merge.exists() {
        return review_merge;
    }
    // Fallback to legacy MERGE_REVIEW.md
    let merge_review = change_dir.join("MERGE_REVIEW.md");
    if merge_review.exists() {
        return merge_review;
    }
    // Fallback to oldest legacy name
    change_dir.join("ARCHIVE_REVIEW.md")
}
